```plantuml
Table(Tag, "Tag (Теги)") {
  tag_id : INT PK
  --
  name: VARCHAR(20) NOT NULL (Название тега)
}

Table(Task_Tag, "Task_Tag (Связка тега и задачи)") {
  task_id: INT PK FK(Task.task_id) (ID задачи)
  tag_id: INT PK FK(Tag.tag_id) (ID тега)
}

Table(Notif, "Notif (Напоминания)") {
  notif_id : INT PK
  --
  task_id: INT PK FK(Task.task_id) (ID задачи)
  notif_time: TIMESTAMP NOT NULL (Дата напоминания)
  repeat_count: INT NOT NULL DEFAULT 1 (Число напоминаний, бесконечно - 0)
  repeat_period: INT NOT NULL DEFAULT 1 (Дней между повторами)
}

Task ||--o{ Task_Tag : "1:M"
Tag ||--o{ Task_Tag : "1:M"
Task ||--o{ Notif : "1:M"
@enduml
```
