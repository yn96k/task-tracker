```plantuml
title "Поиск задачи GET /task?filter="Учеба""

actor user as "Пользователь"

participant apiGateway as "API Gateway"
participant taskService as "Сервис задач"
database taskService_db as "БД сервиса задач"

user -> apiGateway++: Запрос на поиск задачи с ключевым словом
apiGateway -> apiGateway: Аутентификация

apiGateway -> taskService++: Запрос на поиск задачи
taskService -> taskService_db++: Запрос на поиск в базу данных

alt case "Найдены записи по условию"
taskService <- taskService_db: Ответ с записями

apiGateway <- taskService: Ответ с записями
user <- apiGateway: Ответ с записями
else "Не найдено ни одной записи"
taskService <- taskService_db--: Ответ "Не найдено задач"
apiGateway <- taskService: Ответ "Не найдено задач"
user <- apiGateway--: Ответ "Не найдено задач"
end
```
