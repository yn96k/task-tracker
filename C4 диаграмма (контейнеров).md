```plantuml
@startuml "task_tracker_MSA"
!include https://raw.githubusercontent.com/plantuml-stdlib/C4-PlantUML/master/C4_Container.puml

SHOW_PERSON_OUTLINE()

Person(user, "Пользователь", "Манипулирует своими задачами, ставит и получает уведомления")

Container_Ext(push_service, "Сервис push-уведомлений", "")

System_Boundary(c1, "Трекер задач") {
    Container(app, "Веб-приложение", "React JS")
    
    Container(api_gateway, "API Gateway", "nginx + OAuth 2.0")

    Container_Boundary(users_b,"Сервис пользователей") {
        Container(users_sv, "Сервис пользователей", "SpringBoot", "Хранит данные о пользователях, логика регистрации и профилей")
        ContainerDb(users_db, "БД сервиса пользователей", "PostgreSQL")
    }
    Container_Boundary(tasks_b, "Сервис задач") {
        Container(tasks_sv, "Сервис задач", "SpringBoot", "Хранит данные о задачах, срока, тегах и напоминаниях")
        ContainerDb(tasks_db, "БД сервиса задач", "PostgreSQL")
    }
}

Rel_D(users_sv, users_db, "")
Rel_D(tasks_sv, tasks_db, "")

Rel_D(user, app, "вносит изменения в задачи, просматривает задачи, получает уведомления", "https")
BiRel(app, api_gateway, "направляет все запросы и получает ответы, передает данные для аутентификации", "https")

BiRel_D(api_gateway, users_sv, "запрашивает и записывает данные о пользователе", "https")
BiRel_L(api_gateway, tasks_sv, "запрашивает и записывает данные о задачах и уведомлениях", "https")

Rel_R(tasks_sv, push_service, "добавляет в очередь новые уведомления", "https")
Rel_U(push_service, user, "отправляет push-уведомление в ОС смартфона", "https")

@enduml
```
